│                      🚀 INSTALLATION EXPRESS (3 MIN) 🚀                  │
└──────────────────────────────────────────────────────────────────────────┘

1️⃣  COPIER LES FICHIERS
    Copy-Item -Path "SCCM_Monitoring" -Destination "C:\" -Recurse

2️⃣  INITIALISER (remplacer PS1 par votre site code)
    cd C:\SCCM_Monitoring\Scripts
    .\Initialize-Monitoring.ps1 -SiteCode "PS1"

3️⃣  ACTIVER LA COLLECTE HORAIRE
    .\Initialize-Monitoring.ps1 -SiteCode "PS1" -SetupScheduledTask

✅ TERMINÉ ! La collecte démarre automatiquement toutes les heures.

            Hour = ([datetime]$_.Timestamp).Hour
            AvgTime = [double]$_.MetricValue
        }
    } | Group-Object Hour

    # Collecte toutes les 30 minutes au lieu d'1 heure
    $trigger = New-ScheduledTaskTrigger -Once -At (Get-Date) `
        -RepetitionInterval (New-TimeSpan -Minutes 30)
    Set-ScheduledTask -TaskName "SCCM-Monitoring-Hourly" -Trigger $trigger
